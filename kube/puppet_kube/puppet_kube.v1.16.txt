1/ generation du fichier contenant nos vars : 


boogie$ cat env                                                                                   [☸ kube-infra:fso]
OS=redhat
VERSION=1.16.2
CONTAINER_RUNTIME=cri_containerd
CNI_PROVIDER=calico
CNI_PROVIDER_VERSION=3.8
ETCD_INITIAL_CLUSTER=paasdev:192.168.58.177
ETCD_IP="%{::ipaddress_eth0}"
KUBE_API_ADVERTISE_ADDRESS="%{::ipaddress_eth0}"
INSTALL_DASHBOARD=true

docker run --rm -v $(pwd):/mnt --env-file env puppet/kubetool:5.0.0
